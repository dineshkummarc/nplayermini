安装方式：

  • 1. 加入JS文件引用：<script src="js/NPlayerMini.js" type="text/javascript"></script>
  • 2. 加入CSS文件引用：<link rel="stylesheet" href="css/styleMini.css" type="text/css">
  • 3. 直接复制以下代码或在下面的颜色选择器中选择背景颜色后再复制代码：
    var nPlayer = new NPlayer({
      containerId:'nPlayer_mini', //容器的ID（播放器会自动在容器中居中）
      width:300, //播放器的宽度
      height:80, //播放器的高度
      backgroundColor:'#EEE' //播放器的背景颜色
    });
